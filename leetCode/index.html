  if (root) {
    traversePath.pus(root.value);

  }
}

function inOder(root) {
  if (root) {
    inOder(root.left);
    traversePath.pus(root.value);
    inOder(root.right);
  }
}

const traversePath = [];
function postOder(root) {
  if (root) {
    postOder(root.left);
    postOder(root.right);
    traversePath.pus(root.value);
  }
}

# 7. 深度、广度优先搜索

const visited = [];
function dfs(node) {
  // 递归终止条件
  if (node in visited) {
    return;
  }
  // 处理当前层
  visited.push(node);
  // 下探到下一层
  dfs(node.left);
  dfs(node.right);
}

# 8. 贪心算法

  • 贪心
    • 在每一步选择中都采取在当前状态下最好或最优解的选择，从而希望导致结果是全局最好或最优解
  • vs 回溯
    • 能够回退
  • vs 动态规划
    • 最优判断+回退
